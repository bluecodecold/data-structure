## 存储结构
- 顺序存储
- 链式存储

##### 顺序存储
~~~c
#define MAXSIZE 10
struct TreeNode{
    ElementType data; //结构中的数据
    bool isEmpty; //节点是否为空
};
~~~
TreeNode t[MaxSize]; //创建一个数组t，按照从上到下、从左到右的顺序依次存储完全二叉树中的各个节点

***能够看出，只能存储完全二叉树***


##### 链式存储
~~~c
typedef struct BitNode{
    ElementType data;  //结点数据
    struct BitNode *lchild,*rchild; //左右孩子指针
} BitNode,*BitTree;

BitTree t = NULL; //定义一个空树

//插入根节点
root =(BiTree)malloc(sizeof(BitNode));
root->data = 1;
root->lchild = root->rchild = NULL;


//插入新节点
BitNode *p = (BitNode *)malloc(sizeof(BitNode));
p->data = 2;
p->lchild = p->rchild = NULL;
root->lchild = p; //作为根节点的左孩子

~~~

~~~c
typedef struct BitNode{
~~~
