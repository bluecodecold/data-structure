### 单链表的建立
给定多个数据元素（Elemtype），然后将他们组合到一个单链表中

有两步：
1. 初始化一个单链表
2. 每次初始化一个数据元素，插入到表头/表尾
  



#### 普通的插入
~~~c

//在第i个位置插入元素e（通过普通的插法，有头结点）
bool ListInsert(LinkList &L,int i,ElemType e){
    if (i < 1) // i的位置不合法
        return false;
    LNode *p;
    int j = 0;
    p = L;
    while(p != NULL && j < i-1) // 循环找到第i-1个节点
        p = p->next;   //这个循环每次都从头开始之后遍历，时间复杂度为 O(n\^2)
        j++;

    if (p == NULL) // i的位置不合法
        return false;
    LNode *s = (LNode *)malloc(sizeof(LNode));
    s->data = e;  //用s结点保存新数据
    s->next = p->next;  
    p->next = s;  //将s连到p后面
    return true;
}
~~~
#### 尾插法
~~~c
LinkList List_TailInsert(LinkList &L)
{ 
    L = (LNode *)malloc(sizeof(LNode)); //建立头结点
    LNode *s, *r = L;  // r为指向表尾的指针
    scanf("%d", &x); //输入结点的值
    while(x != 9999) //输入9999表示结束
    {
        s = (LNode *)malloc(sizeof(LNode));  
        s->data = x; //用s结点保存新数据
        r->next = s;  //在r结点之后插入一个元素x 
        r = s; //r指向新的s结点
        scanf("%d", &x);
    }
    r->next = NULL;  //将表尾结点置空
    return L;  //返回头指针
}
~~~

#### 头插法
~~~c
LinkList List_HeadInsert(LinkList &L){
    LNode *s;
    int x;
    L =(LinkList)malloc(sizeof(LNode));
    L->next = NULL;  //只要初始化单链表！都要先把头指针指向NULL！
    scanf("%d", &x);
    while(x != 9999) //输入9999表示结束
    {
        s = (LNode *)malloc(sizeof(LNode));
        s->data = x;
        s->next = L->next;
        L->next = s;
        scanf("%d", &x);
    }
    return L;
}
~~~
