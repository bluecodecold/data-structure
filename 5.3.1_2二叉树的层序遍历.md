## 算法思想：
1. 初始化一个辅助队
2. 根结点入队
3. 若队列非空，则队头结点出队，访问该结点，并将其左、右孩子插入队尾（如果有的话）
4. 重复
5. 直至队列为空

~~~c

typedef struct BiTNode
{
    char data;
    struct BiTNode *lchild,*rchild;
}BiTnode,*BiTree;

typedef struct LinkNode{  //辅助队列的结点
    BiTNode *data;  //存指针而不是结点
    struct LinkNode *next;
}LinkNode;

typedef struct{  //辅助队列的声明
    LinkNode *front,*rear; //队头指针，队尾指针
}LinkQueue;

void LevelOrder(BitTree T)
{
    LinkQueue Q; 
    InitQueue(Q); //初始化队列
    BiTree p;  
    EnQueue(Q,T);  //将根节点入队
    while(!QueueEmpty(Q)) //队列不空就循环 
        Dequeue(Q,p); 队头结点出队
        visit(p); 访问出队结点
        if(p->lchild!=NULL)
            EnQueue(Q,p->lchild);//左孩子入队
        if(p->rchild!=NULL)
            EnQueue(Q,p->rchild);//右孩子入队
}
~~~
